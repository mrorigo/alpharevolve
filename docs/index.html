config.ts 
const config = {
  // Core problem definition
  initialSolution: initialCode,
  fitnessFunction: evaluateCandidate,
  
  // Evolution settings
  generations: 10,
  populationSize: 5,
  
  // AI Configuration
  llmModel: 'gemini-1.5-pro', 
  feedbackEnabled: true,
  
  // Safety
  sandbox: { memoryLimit: 512, timeout: 5000 }
};

const evolver = new AlphaRevolve(config);
await evolver.run();
